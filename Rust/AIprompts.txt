use std::fs;

fn main() {
    let paths = fs::read_dir("./images").unwrap();

    let mut image_paths = Vec::new();

    for path in paths {
        let path_str = path.unwrap().path().display().to_string();
        if path_str.ends_with(".png") {
            image_paths.push(path_str);
        }
    }

    println!("{:?}", image_paths);
}
